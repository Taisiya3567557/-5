import psycopg2

def get_courses_without_enrollments():
    # Подключаемся к базе PostgreSQL
    conn = psycopg2.connect(
        dbname="your_db_name",
        user="your_username",
        password="your_password",
        host="your_host",
        port="your_port"
    )
    cursor = conn.cursor()

    # SQL-запрос для нахождения курсов без регистраций
    query = """
    SELECT c.course_id, c.course_title
    FROM Courses c
    LEFT JOIN Enrollments e ON c.course_id = e.course_id
    WHERE e.course_id IS NULL;
    """
    
    cursor.execute(query)
    courses = cursor.fetchall()

    conn.close()
    return courses

if __name__ == "__main__":
    courses = get_courses_without_enrollments()
    
    if courses:
        print("Курсы без регистраций:")
        for course_id, course_title in courses:
            print(f"{course_id}: {course_title}")
    else:
        print("Все курсы имеют регистрации.")
